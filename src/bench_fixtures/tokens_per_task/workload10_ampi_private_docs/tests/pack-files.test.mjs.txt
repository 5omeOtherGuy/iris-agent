import assert from 'node:assert/strict';
import test from 'node:test';
import { publishedFiles } from '../src/pack-files.mjs';

test('excludes docs/private from the published file list', () => {
  assert.deepEqual(
    publishedFiles(['README.md', 'docs/public.md', 'docs/private/operator-notes.md', 'src/index.ts']),
    ['README.md', 'docs/public.md', 'src/index.ts'],
  );
});

test('keeps ordinary docs and source files', () => {
  assert.deepEqual(publishedFiles(['docs/README.md', 'src/index.ts']), ['docs/README.md', 'src/index.ts']);
});
