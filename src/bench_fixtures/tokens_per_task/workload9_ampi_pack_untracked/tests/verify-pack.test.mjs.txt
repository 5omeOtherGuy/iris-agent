import assert from 'node:assert/strict';
import test from 'node:test';
import { assertPublishable } from '../src/verify-pack.mjs';

test('blocks publish when npm dry-run reports untracked files', () => {
  assert.throws(
    () => assertPublishable([{ path: 'src/index.ts' }, { path: 'debug.log', untracked: true }]),
    /untracked files in package: debug\.log/,
  );
});

test('keeps clean tarballs publishable', () => {
  assert.equal(assertPublishable([{ path: 'src/index.ts' }, { path: 'README.md' }]), true);
});
