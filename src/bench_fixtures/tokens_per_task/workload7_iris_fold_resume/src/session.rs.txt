#[derive(Debug, Clone, PartialEq, Eq)]
pub struct Entry {
    pub kind: &'static str,
    pub id: u32,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct ResumeState {
    pub last_id: u32,
    pub next_seq: u32,
}

/// BUG (mirrors Iris PR #394): persisted `fold` entries are leaf-chain entries,
/// but resume ignores them, so the next append can reuse the fold id and parent
/// around it.
pub fn scan_for_resume(entries: &[Entry]) -> ResumeState {
    let mut state = ResumeState { last_id: 0, next_seq: 1 };
    for entry in entries {
        match entry.kind {
            "message" | "compaction" | "modelSelection" => {
                state.last_id = entry.id;
                state.next_seq = state.next_seq.max(entry.id + 1);
            }
            _ => {}
        }
    }
    state
}

pub fn append_message_after_resume(entries: &[Entry]) -> Entry {
    let state = scan_for_resume(entries);
    Entry { kind: "message", id: state.next_seq }
}
