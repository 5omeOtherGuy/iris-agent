use recall_span_fixture::recall::RecallStore;

#[test]
fn standalone_span_without_handle_returns_original_turns() {
    let store = RecallStore::sample();
    let text = store
        .recall(None, Some("00000001"), Some("00000002"))
        .expect("standalone span should recall from this session");
    assert!(text.contains("inspect parser.rs"));
    assert!(text.contains("read parser.rs"));
}

#[test]
fn out_of_range_standalone_span_is_an_error() {
    let store = RecallStore::sample();
    let err = store
        .recall(None, Some("00000010"), Some("00000020"))
        .expect_err("empty explicit spans must be tool errors");
    assert!(err.contains("selected no turns"), "unexpected error: {err}");
}
