import assert from 'node:assert/strict';
import test from 'node:test';
import { githubToken } from '../src/auth.mjs';

test('accepts GH_TOKEN as a GitHub auth fallback', () => {
  assert.equal(githubToken({ GH_TOKEN: 'gh-token' }), 'gh-token');
});

test('accepts GITHUB_PERSONAL_ACCESS_TOKEN as a GitHub auth fallback', () => {
  assert.equal(githubToken({ GITHUB_PERSONAL_ACCESS_TOKEN: 'pat-token' }), 'pat-token');
});

test('keeps existing precedence', () => {
  assert.equal(githubToken({ AMPI_GITHUB_TOKEN: 'ampi', GH_TOKEN: 'gh' }), 'ampi');
});
