use crate::reasoning::ReasoningEffort;

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct CodexRequest {
    pub model: &'static str,
    pub input: Vec<&'static str>,
    pub reasoning: Option<CodexReasoning>,
}

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct CodexReasoning {
    pub effort: &'static str,
    pub summary: Option<&'static str>,
}

pub fn build_request(reasoning: Option<ReasoningEffort>) -> CodexRequest {
    CodexRequest {
        model: "gpt-5.4",
        input: Vec::new(),
        reasoning: codex_reasoning(reasoning),
    }
}

/// Map a normalized reasoning level to the Codex Responses reasoning object.
///
/// BUG (mirrors PR #404): the request sends only `effort`, so the Responses
/// API computes reasoning but streams no live reasoning summary deltas. The
/// adapter must also request `summary: "auto"` when reasoning is enabled.
fn codex_reasoning(reasoning: Option<ReasoningEffort>) -> Option<CodexReasoning> {
    let effort = reasoning?.as_codex_effort()?;
    Some(CodexReasoning {
        effort,
        summary: None,
    })
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn reasoning_off_omits_reasoning_block() {
        let request = build_request(Some(ReasoningEffort::Off));
        assert_eq!(request.reasoning, None);
    }
}
